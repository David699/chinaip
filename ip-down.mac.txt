#!/bin/bash

export LC_ALL=C
export PATH="$PATH:/sbin:/usr/sbin:/bin:/usr/bin"

MAGIC=3c3bd1da74bfa2379ecb01dab861f829
sed -n -e "/^$MAGIC/,\$p" $0 | while read line; do
	if [[ $line =~ ^[0-9.]+/[0-9]+$ ]]; then
		route delete $line
	fi
done
exit
